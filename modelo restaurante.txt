	dados numMesa, kg, qtdSobremesa, qtdRefri, qtdCerv, total, numPessoas, totalPessoas;	valKg = 30.00; valSobremesa = 6.00; valRefri = 5.00; valCerv = 10.00;

	
	OP. adcNumPessoas(num)
	 inicio
		 numPessoas = numPessoas + num;

	 inicio
		 Kg = Kg + peso;

	 inicio
		 qtdSobremesa = qtdSobremesa + qtd;

	 inicio

	 fim

	 inicio

	 fim

	 fim
